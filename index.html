"pieno", "fuori servizio", "ottima posizione") create table comments ( id bigint generated always as identity primary key, locker_id text not null, -- id del locker nell'app (es. "way/123456", "inpost/ITABB0123") autore text, -- nome facoltativo di chi scrive testo text, foto_url text, -- link alla foto, se presente status text not null default 'pending', created_at timestamptz not null default now() ); alter table comments enable row level security; create policy "leggi solo commenti approvati" on comments for select using (status = 'approved'); create policy "chiunque può commentare" on comments for insert with check (status = 'pending'); -- Archivio foto: bucket pubblico per le immagini caricate dagli utenti insert into storage.buckets (id, name, public) values ('locker-photos', 'locker-photos', true); create policy "chiunque può caricare una foto" on storage.objects for insert with check (bucket_id = 'locker-photos'); create policy "foto sempre visibili (il commento resta comunque moderato)" on storage.objects for select using (bucket_id = 'locker-photos');
